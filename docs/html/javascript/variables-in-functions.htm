Variables in functions.

Define all variables at the top of the function:


function foo() {
	var a = 1,
		b = 'something',
		c = false,
		e, d, g,
		a;
	
	// no var statement after this point
}

Try to define variables in the order they are used:


function foo() {
	var a = true,
		b = 2;
		
	if (a) {
		check(b);
	}
}

Try to give as much information as you can about the flow of the function, by the way you order/group/initialize the variables:


function foo() {
	var a = 2,  // seperate variable
		b = 3,  // seperate variable
		c, d,   // these two are probably related
		d,		// you don't know what the type will be
		e = '', // you know the type will be string
		f = {}, // you know the type will be object
		g = []; // you know the type will be array
}

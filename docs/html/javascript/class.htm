Class

Avoid using constructor functions that are instantiated with the new keyword and named using PascalCase naming convention.


// Don't do this:
function Person(name, age) {
	this.name = name || 'John';
	this.age  = age  || 20;;
}
var ali = new Person('ali', 26);

Use regular functions as constructors, and prefix their names with the letters `new` (newPerson).


// do this instead:
function newPerson(name, age) {
	var inst = {};
	
	inst.name = name || 'John';
	inst.age  = age  || 20;
	
	return inst;
}
var ali = newPerson('ali', 26);

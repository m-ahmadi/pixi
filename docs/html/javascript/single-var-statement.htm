Single Var Statement.

  • Use single var statements.
  • Do not use multiple var statements unless you have a good excuse.

function foo() {
	var a, b, c,
		d, e, f,
		g = 1,
		h = '.png',
		i = false,
		j = 0x4400FF;
}

You can see the above code is easily gets cluttered and harder to read with multiple var statements:


function foo() {
	var a, b, c;
	var	d, e, f;
	var	g = 1;
	var	h = '.png';
	var	i = false;
	var	j = 0x4400FF;
}
